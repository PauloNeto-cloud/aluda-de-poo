// Classe que representa uma conta bancária

public class ContaBancaria {

    // Atributo privado (não pode ser acessado diretamente de fora)

    private double saldo;

    // Construtor: define o saldo inicial

    public ContaBancaria(double saldoInicial) {
        this.saldo = saldoInicial;
    }

    // Getter: método que devolve o valor do saldo

    public double getSaldo() {
        return saldo;
    }

    // Setter: método que altera o saldo (aqui fazemos depósito)

    public void depositar(double valor) {
        // Só permite depósito de valores positivos

        if (valor > 0) {
            saldo += valor; // soma ao saldo atual
        }
    }
}

public class Main {
    public static void main(String[] args) {
        // Criando uma conta com R$100 de saldo inicial

        ContaBancaria conta = new ContaBancaria(100);

        // Fazendo um depósito de R$50
        
        conta.depositar(50);

        // Consultando o saldo (via getter)

        System.out.println("Saldo: R$" + conta.getSaldo());

        // Saída: Saldo: R$150.0
    }
}
