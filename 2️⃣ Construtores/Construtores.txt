// Classe Pessoa com construtor

public class Pessoa {
    String nome;
    int idade;

    // Construtor (inicializa o objeto já com valores)

    public Pessoa(String nome, int idade) {

        // "this" se refere ao atributo da classe

        this.nome = nome;
        this.idade = idade;
    }

    // Método para exibir informações

    void falar() {
        System.out.println(nome + " tem " + idade + " anos.");
    }
}

public class Main {
    public static void main(String[] args) {

        // Criando objeto com construtor (passando valores direto)

        Pessoa p1 = new Pessoa("Paulo", 19);

        // Chamando o método
        // Saída: Paulo tem 19 anos.
        
        p1.falar();
    }
}
