// Classe Pai (superclasse)

public class Pessoa {
    String nome;
    int idade;

    // Método da classe pai

    void apresentar() {
        System.out.println("Meu nome é " + nome);
    }
}

// Classe Filha (subclasse) que herda de Pessoa

public class Aluno extends Pessoa {
    // Atributo extra, só existe no aluno

    String curso;

    // Método da classe filha

    void estudar() {
        System.out.println(nome + " está estudando " + curso);
    }
}

public class Main {
    public static void main(String[] args) {

        // Criando um aluno (que é uma Pessoa também)

        Aluno a1 = new Aluno();

        // Definindo valores herdados de Pessoa

        a1.nome = "Paulo";
        a1.idade = 19;

        // Definindo valor específico de Aluno

        a1.curso = "Ciência da Computação";

        // Chamando método herdado

        a1.apresentar(); // Saída: Meu nome é Paulo

        // Chamando método da própria classe Aluno
        
        a1.estudar();    // Saída: Paulo está estudando Ciência da Computação
    }
}
